            bPreview: true,
            roomId,
            width: 640,
            height: 360,
            componentId: 0
        },
        setMETemplateData: {
            switcherId: '123',
            action: 'init',
            type: 1,
            templateId: '123',
            templateName: 'xxx',
            templateInfo: [],
            backgroundUrl: ''
        },

    });

}
